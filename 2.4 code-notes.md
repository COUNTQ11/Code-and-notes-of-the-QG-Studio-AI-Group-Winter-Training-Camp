#集合运算
A = {1, 2, 3, 4}
B = {3, 4, 5, 6}
#交集(获得两个集合共有的部分)

#并集(获得两个集合合并后的结果)
print(A | B)
#差集(X-Y代表获得A中有的，B中没有的元素)
print(A - B)
#对称差集(获取双方非重叠部分的元素，X^Y是(X-Y)|(Y-X))
print(A ^ B)
#判断子集
print(A <= B)#判断A是否是B的子集，输出布尔类型
